// TESTE DE DIAGNÓSTICO DO FIREBASE console.log("Iniciando teste do Firebase..."); const testarFirebase = async () => { try { // 1. Tenta escrever um dado await db.collection("teste_conexao").add({ mensagem: "Conexão OK", timestamp: new Date() }); console.log("✅ SUCCESSO: Firestore gravou dados com sucesso!"); // 2. Tenta ler esse dado const snapshot = await db.collection("teste_conexao").get(); console.log(`✅ SUCCESSO: Firestore leu ${snapshot.size} documentos.`); } catch (error) { console.error("❌ ERRO NO FIREBASE:"); console.error("Código do erro:", error.code); console.error("Mensagem:", error.message); // Dica visual na tela para você saber na hora document.body.innerHTML += `
ERRO NO FIREBASE: ${error.message}
`; } }; testarFirebase();
